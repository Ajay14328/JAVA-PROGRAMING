import java.util.Scanner; 
public class linsrc
{
 public static void main(String[] args)
{
   int[] arr={1,2,3,2,2,5};
   int target=5;
   Boolean res=false;
   for(int i=0;i<arr.length;i++){
           if(arr[i]==target){
                res=true;
               System.out.println("element found"); 
            break;
       } 
     }
if(res==false){
    System.out.println("not found");
}
}
}
